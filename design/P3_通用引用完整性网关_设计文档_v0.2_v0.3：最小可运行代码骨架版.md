5）logic_fingerprint/p3_reference_gateway/parser.py

先做最小规则版，只支持形如 [ref:abc123] 的引用格式。
这样最稳，不会一开始就把解析器做复杂。

from __future__ import annotations

import re

from .models import ReferenceClaim

REFERENCE_PATTERN = re.compile(r"\[ref:([A-Za-z0-9_\-:.]+)\]")


def parse_reference_claims(llm_output: str) -> list[ReferenceClaim]:
    """
    最小版解析器：
    仅解析 [ref:<member_id>] 格式，例如：
    [ref:doc_1]
    [ref:chunk-002]
    """
    claims: list[ReferenceClaim] = []

    for match in REFERENCE_PATTERN.finditer(llm_output):
        raw_text = match.group(0)
        ref_id = match.group(1)
        position = match.start()
        claims.append(
            ReferenceClaim(
                raw_text=raw_text,
                ref_id=ref_id,
                position=position,
            )
        )

    return claims
6）logic_fingerprint/p3_reference_gateway/resolver.py
from __future__ import annotations

from .models import (
    ContextPool,
    ReferenceClaim,
    ResolutionResult,
    ResolvedReference,
    UnresolvedReference,
)


def resolve_reference_claims(
    claims: list[ReferenceClaim],
    context_pool: ContextPool,
) -> ResolutionResult:
    """
    当前最小版规则：
    - ref_id 为空 -> unresolved(parse_failed)
    - ref_id 以 outside: 开头 -> outside_pool
    - ref_id 以 fake: 开头 -> fabricated
    - ref_id 命中 ContextPool.member_id -> resolved
    - 否则 -> unresolved(id_not_found)
    """
    member_map = {member.member_id: member for member in context_pool.members}

    resolved: list[ResolvedReference] = []
    unresolved: list[UnresolvedReference] = []
    outside_pool_references: list[str] = []
    fabricated_references: list[str] = []

    for claim in claims:
        if not claim.ref_id:
            unresolved.append(
                UnresolvedReference(raw_text=claim.raw_text, reason="parse_failed")
            )
            continue

        if claim.ref_id.startswith("outside:"):
            outside_pool_references.append(claim.ref_id)
            continue

        if claim.ref_id.startswith("fake:"):
            fabricated_references.append(claim.ref_id)
            unresolved.append(
                UnresolvedReference(raw_text=claim.raw_text, reason="fabricated")
            )
            continue

        member = member_map.get(claim.ref_id)
        if member is not None:
            resolved.append(ResolvedReference(ref_id=claim.ref_id, member=member))
        else:
            unresolved.append(
                UnresolvedReference(raw_text=claim.raw_text, reason="id_not_found")
            )

    return ResolutionResult(
        resolved=resolved,
        unresolved=unresolved,
        outside_pool_references=outside_pool_references,
        fabricated_references=fabricated_references,
    )

from __future__ import annotations

from .constants import (
    PLUGIN_NAME,
    SEVERITY_CRITICAL,
    SEVERITY_HIGH,
    SEVERITY_LOW,
    SEVERITY_MEDIUM,
    STATUS_CLEAN,
    STATUS_FABRICATED,
    STATUS_OUTSIDE_POOL,
    STATUS_PARTIAL,
    STATUS_UNRESOLVED,
    TRIGGER_CITATION_OUTSIDE_CONTEXT_POOL,
    TRIGGER_FABRICATED_REFERENCE_DETECTED,
    TRIGGER_REFERENCE_MEMBER_NOT_FOUND,
    TRIGGER_REFERENCE_PARSE_FAILED,
    TRIGGER_UNRESOLVED_REFERENCE_POINTER,
)
from .contracts import P3Result
from .models import ResolutionResult


def evaluate_resolution_result(resolution: ResolutionResult) -> P3Result:
    resolved_count = len(resolution.resolved)
    unresolved_count = len(resolution.unresolved)
    outside_count = len(resolution.outside_pool_references)
    fabricated_count = len(resolution.fabricated_references)

    triggers: list[str] = []
    reasons: list[str] = []

    if outside_count > 0:
        triggers.append(TRIGGER_CITATION_OUTSIDE_CONTEXT_POOL)
        reasons.append("Detected references that are outside the current context pool.")
        status = STATUS_OUTSIDE_POOL
        severity = SEVERITY_CRITICAL

    elif fabricated_count > 0:
        triggers.append(TRIGGER_FABRICATED_REFERENCE_DETECTED)
        reasons.append("Detected fabricated references.")
        status = STATUS_FABRICATED
        severity = SEVERITY_CRITICAL

    elif resolved_count > 0 and unresolved_count == 0:
        status = STATUS_CLEAN
        severity = SEVERITY_LOW
        reasons.append("All references were resolved inside the current context pool.")

    elif resolved_count > 0 and unresolved_count > 0:
        status = STATUS_PARTIAL
        severity = SEVERITY_MEDIUM
        triggers.append(TRIGGER_UNRESOLVED_REFERENCE_POINTER)
        triggers.append(TRIGGER_REFERENCE_MEMBER_NOT_FOUND)
        reasons.append("Some references were resolved, but some could not be resolved.")

    elif unresolved_count > 0:
        status = STATUS_UNRESOLVED
        severity = SEVERITY_HIGH
        triggers.append(TRIGGER_UNRESOLVED_REFERENCE_POINTER)

        unresolved_reasons = {item.reason for item in resolution.unresolved}
        if "parse_failed" in unresolved_reasons:
            triggers.append(TRIGGER_REFERENCE_PARSE_FAILED)
        if "id_not_found" in unresolved_reasons:
            triggers.append(TRIGGER_REFERENCE_MEMBER_NOT_FOUND)

        reasons.append("No references could be reliably resolved.")

    else:
        # 没有任何引用时，当前最小版按 clean 处理
        status = STATUS_CLEAN
        severity = SEVERITY_LOW
        reasons.append("No explicit references were found in the output.")

    return P3Result(
        plugin_name=PLUGIN_NAME,
        reference_integrity_status=status,
        resolved_members=[item.ref_id for item in resolution.resolved],
        unresolved_members=[item.raw_text for item in resolution.unresolved],
        outside_pool_references=resolution.outside_pool_references,
        severity=severity,
        triggers=triggers,
        reasons=reasons,
        metrics={
            "resolved_count": resolved_count,
            "unresolved_count": unresolved_count,
            "outside_pool_count": outside_count,
            "fabricated_count": fabricated_count,
        },
    )

from __future__ import annotations

from .contracts import HostP3Request, P3Result
from .evaluator import evaluate_resolution_result
from .parser import parse_reference_claims
from .resolver import resolve_reference_claims


class ReferenceIntegrityGateway:
    """
    P3 最小总调度器：
    HostP3Request -> parse -> resolve -> evaluate -> P3Result
    """

    def run(self, request: HostP3Request) -> P3Result:
        claims = parse_reference_claims(request.llm_output)
        resolution = resolve_reference_claims(claims, request.context_pool)
        result = evaluate_resolution_result(resolution)
        return result

def build_pool() -> ContextPool:
    return ContextPool(
        pool_id="pool_1",
